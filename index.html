@app.get("/validate", response_model=ValidationResponse)

$ uvicorn nhs_number_api.main:app --host 0.0.0.0 --port 8000

step 2: create a container


FROM python:3.11-slim
WORKDIR /app
COPY pyproject.toml ./
COPY src ./src
CMD ["uvicorn", "nhs_number_api.main:app", "--host", "0.0.0.0", "--port", "8888"]
                    

$ docker build -f Containerfile -t nhs-number-api:v0.0.1 .
                    

$ docker run --rm -p 8888:8888 --read-only nhs-number-api:v0.0.1
                    

$ docker push ghcr.io/amadeuspzs/nhs-number-api/nhs-number-api:v0.0.1

step 3: deploy the container into a kubernetes (k8s) cluster


apiVersion: apps/v1
kind: Deployment
metadata:
  name: nhs-number-api
spec:
  replicas: 1
  template:
    spec:
      containers:
        - name: nhs-number-api
          image: ghcr.io/amadeuspzs/nhs-number-api/nhs-number-api
          tag: v0.0.1
          ports:
            - containerPort: 8888
              name: http

apiVersion: v1
kind: Service
metadata:
  name: nhs-number-api
spec:
  type: ClusterIP
  selector:
    app: nhs-number-api
  ports:
    - name: http
      port: 80
      targetPort: 8888
      protocol: TCP


apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: nhs-number-api
spec:
  rules:
    - host: nhs-rpysoc.rsna.newtonstree.ai
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: nhs-number-api
                port:
                  name: http
  tls:
    - hosts:
        - nhs-rpysoc.rsna.newtonstree.ai
